

加item界面

localhost:9000/additem
methods=["POST"]

incoming request format: Json
sellerID:int
name:string
category:string
quantity:int
cur_price:float
price_step:float
start_time:datetime
end_time:datetime
buy_now:bool
buy_now_price:int
shipping_cost:int
description:string
photo:string (base64)

return request format: Json
{"success":False,"message":"sellerID is invalid"} //blocked user
{"success":False,"message":"Category Not Exists"} 
{"success":True,"message":1} //itemid

methods=["GET"]

incoming request format: None
None

{"success":True,"message":{ID1:Name1,ID2:Name2,ID3:Name3,...}}
ID:int
Name:string


通过category获取对应商品
localhost:9000/searchbycategory
methods=["GET"]

incoming request format: json
{"categoryID":int}

return request format: json
{"success":True,"message":{ID1:(name,price,url,photo)},ID2:(name,price,url,photo),...}}
ID:int
item_info:(,,,) tuple of info


通过名字搜索商品
localhost:9000/searchbyname
 methods=["GET"]

incoming request format: json
{"name":string}

return request format: json
{"success":False} //not found name
{"success":True,"message":url}
url:string
